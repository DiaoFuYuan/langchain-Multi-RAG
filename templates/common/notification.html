通知组件演示

基本用法

// 显示成功通知
Notification.success('操作成功完成！', '成功', 3000);

// 显示错误通知
Notification.error('操作发生错误！', '错误', 3000);

// 显示警告通知
Notification.warning('请注意此操作的风险！', '警告', 3000);

// 显示信息通知
Notification.info('系统将于今晚进行维护。', '提示', 3000);

// 清除所有通知
Notification.clear();

高级用法

// 自定义持续时间（5秒）
Notification.success('操作成功完成！', '成功', 5000);

// 带回调函数
Notification.info('点击确定继续', '提示', 3000, function() {
    console.log('通知已关闭');
    alert('通知关闭了！');
});

// 不自动关闭（持续时间设为0）
Notification.warning('这条通知不会自动关闭', '警告', 0);
